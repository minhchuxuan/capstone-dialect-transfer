\begin{frame}{The problem}
  \begin{itemize}
    \item Vietnamese has three dialect regions: \hl{Northern}, \hl{Central}, \hl{Southern}.
    \item Example (Central): \emph{``Anh ─æi \hl{m├┤} rß╗ōi?''} $=$ standard \emph{``Anh ─æi \hl{─æ├óu} rß╗ōi?''} (``Where did he go?'').
  \end{itemize}
  \vspace{0.4em}

    \node[draw,rounded corners,fill=oiorange!15,minimum width=2.6cm,minimum height=1cm,align=center,right=4.2cm of std] (dia) {Dialectal\\text};

        node[below,font=\scriptsize,text=oired]{\textbf{generation} ŌĆö \emph{novel, this work}} (dia.-150);
  \end{tikzpicture}
  \vspace{0.3em}
  \begin{block}{Our goal}
    One model, both directions \emph{and} three regions --- normalization \textbf{and} the novel
    region-conditioned dialect \emph{generation}.
  \end{block}
\end{frame}

% =============================================================================
\begin{frame}{Contributions}
  \begin{enumerate}
    \item \hl{Novel task}: region-conditioned standard$\rightarrow$dialect generation (Feature~C) for all three regions.
    \item \hl{Unified multi-task model}: one BARTpho-syllable does (A) dialect$\rightarrow$std,
          (B) lexical normalization, (C) std$\rightarrow$dialect --- selected by a text task prefix.
    \item \hl{Evaluation suite for one-to-many}: Dialect Feature Recall, marker-independent
          Dialectal Edit-Recall, region accuracy, copy-rate (BLEU alone is misleading here).
    \item \hl{Data strategy}: oversampling $+$ sampling back-translation for the data-starved regions.
  \end{enumerate}
\end{frame}

\begin{frame}{Background: conditional sequence-to-sequence}
  All three tasks are one conditional generation problem. Source $x$, condition $c$
  (the task/region prefix), target $y$:
  \[
    P_\theta(y \mid x, c) = \prod_{t=1}^{n} P_\theta\!\left(y_t \mid y_{<t},\, x,\, c\right).
  \]
  \begin{itemize}
    \item \hl{Textual conditioning}: input string is \texttt{\{task\}: \{source\}} --- no new heads or embeddings.
    \item \hl{Training}: teacher forcing $+$ token cross-entropy, with \hl{label smoothing}
          (several outputs are acceptable in one-to-many).
    \item \hl{Decoding}: \emph{beam search} for the deterministic normalization direction;
          \emph{sampling} for generation \emph{and} for back-translation (avoids over-regular,
          mode-collapsed outputs --- Edunov et al.\ 2018).
  \end{itemize}
\end{frame}

\begin{frame}{Method (2): reverse corpus + sampling back-translation}
  \centering
  \begin{tikzpicture}[font=\scriptsize,node distance=5mm and 8mm,

    \node[box,fill=oiblue!15,right=of rd] (am) {A-model\\dialect$\to$std\\(\emph{sample})};
    \node[box,fill=oigray!12,right=of am] (ss) {synthetic\\\textbf{standard} (source)};
    \node[box,fill=oigreen!15,right=of ss,minimum width=2.4cm] (pair) {train pair for C:\\(synth-std, real-dia)};
    \draw[-{Latex[length=2mm]}] (rd) -- (am);
    \draw[-{Latex[length=2mm]}] (am) -- (ss);
    \draw[-{Latex[length=2mm]}] (ss) -- (pair);
    \draw[-{Latex[length=2mm]},oiorange,thick] (rd.south) to[bend right=25] (pair.south);
  \end{tikzpicture}
  \vspace{0.3em}
  \begin{itemize}\small
    \item \hl{Reverse corpus}: swap (dialect, standard) $\rightarrow$ (standard, dialect) $+$ region prefix.
          Target side stays \emph{human-authored}.
    \item \hl{Sampling back-translation} (Edunov et al.\ 2018): keep \emph{real dialect} as target,
          \emph{sample} synthetic standard sources from the strong A-model --- sampling, not beam, to avoid mode collapse.
    \item \hl{Rebalancing}: oversample Northern/Southern; light quality filter (round-trip, length ratio, $\geq$1 marker).
  \end{itemize}
\end{frame}

% =============================================================================
\begin{frame}{Method (3): transfer-aware model selection}
  \begin{block}{Problem}
    Pooled validation BLEU rewards a \emph{copying} checkpoint --- copying scores high on
    overlap for std$\rightarrow$dialect.
  \end{block}
  \begin{block}{Fix}
    Select on a \hl{macro per-task BLEU $+$ Dialectal Edit-Recall} composite, so a checkpoint
    is preferred only if it actually performs the dialectal substitutions.
  \end{block}
\end{frame}

% =============================================================================
\begin{frame}{Evaluation metrics}
  \begin{itemize}\small
    \item \textbf{Overlap/semantic}: BLEU, chrF, ROUGE-L, METEOR, BERTScore;
          \textbf{normalization}: WER/CER (A), ERR (B).
    \item \textbf{Dialect Feature Recall (DFR)} --- fraction of the reference's regional markers reproduced:
      \[
        \mathrm{DFR}=\frac{\bigl|\{m\in R: m\in\hat y\}\bigr|}{|R|},\quad R=\{m\in M_r: m\in y\}.
      \]
    \item \textbf{Dialectal Edit-Recall} (marker-independent) --- did the model make the gold's token changes?
      \[
        \mathrm{Recall}=\frac{|A_{\text{gold}}\cap A_{\text{sys}}|}{|A_{\text{gold}}|},\quad
        A_\bullet = T(\cdot)\setminus T(x).
      \]
      Copy $\Rightarrow$ Edit-Recall $=0$, regardless of BLEU.
    \item \textbf{Region accuracy} $+$ \textbf{copy-rate}: controllability and under-generation.
  \end{itemize}
\end{frame}

\begin{frame}{Why syllable tokenization: a probe}
  \centering
  \footnotesize
  \begin{tabular}{l l}
    \toprule

    \midrule
    ``Anh ß║źy ─æi ─æ├óu rß╗ōi?'' (std)  & \texttt{\_Anh \_ß║źy \_─æi \_─æ├óu \_rß╗ōi ?}\\
    ``ß║ónh ─æi m├┤ rß╗ōi?'' (dialect)  & \texttt{\_ß║ónh \_─æi \_m├┤ \_rß╗ōi ?}\\
    ``B├Ā ß║źy n├│i g├¼ vß║Ły?'' (std)   & \texttt{\_B├Ā \_ß║źy \_n├│i \_g├¼ \_vß║Ły ?}\\

    ``Kh├┤ng biß║┐t'' (std)          & \texttt{\_Kh├┤ng \_biß║┐t}\\

  \end{tabular}
  \vspace{0.5em}
  \begin{itemize}\small

    \item No hard segmentation failure on OOV dialect/teencode --- exactly why we pick BARTpho-syllable.
  \end{itemize}
\end{frame}

\begin{frame}{Analysis \& error taxonomy}
  \begin{itemize}
    \item \hl{Under-transfer / copying} (copy-rate, Edit-Recall) --- worst on Northern.
    \item \hl{Wrong-region leakage} (region accuracy / confusion matrix) --- markers of another region appear.
    \item \hl{Over-transfer} (Edit-Precision) --- spurious markers.
    \item \hl{Semantic drift} (BERTScore / human eval).
    \item One-to-many \hl{diversity}: sampling yields varied valid realizations
          (Self-BLEU: Central 81.6, Southern 86.7, Northern 93.7; lower $=$ more diverse).
  \end{itemize}
\end{frame}

% =============================================================================
\begin{frame}{Limitations \& future work}
  \begin{itemize}
    \item Single dialectal reference per sentence; 63 provinces collapsed to 3 regions.
    \item DFR depends on a marker list (mitigated by marker-independent Edit-Recall).
    \item Northern remains smallest; back-translation sources are synthetic.
    \item Future: human evaluation at scale, ViT5 comparison, iterative back-translation rounds.
  \end{itemize}
\end{frame}

    \item Normalization competitive with a much larger SOTA model.
    \item New transfer-aware metrics expose what BLEU hides; data strategy fixes regional imbalance.

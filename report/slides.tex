\begin{frame}{The problem}
  \begin{itemize}
    \item Vietnamese has three dialect regions: \hl{Northern}, \hl{Central}, \hl{Southern}.
    \item Example (Central): \emph{``Anh đi \hl{mô} rồi?''} $=$ standard \emph{``Anh đi \hl{đâu} rồi?''} (``Where did he go?'').
  \end{itemize}
  \vspace{0.4em}
  \centering
  \begin{tikzpicture}[font=\small,node distance=8mm]
    \node[draw,rounded corners,fill=oiblue!12,minimum width=2.6cm,minimum height=1cm,align=center] (std) {Standard\\Vietnamese};
    \node[draw,rounded corners,fill=oiorange!15,minimum width=2.6cm,minimum height=1cm,align=center,right=4.2cm of std] (dia) {Dialectal\\text};
    \draw[-{Latex[length=2.2mm]},oigray,thick] (dia.150) to[bend right=18]
        node[above,font=\scriptsize,text=oigray]{\textbf{normalization} — all prior work} (std.30);
    \draw[-{Latex[length=2.2mm]},oired,very thick] (std.-30) to[bend right=18]
        node[below,font=\scriptsize,text=oired]{\textbf{generation} — \emph{novel, this work}} (dia.-150);
  \end{tikzpicture}
  \vspace{0.3em}
  \begin{block}{Our goal}
    One model, both directions \emph{and} three regions --- normalization \textbf{and} the novel
    region-conditioned dialect \emph{generation}.
  \end{block}
\end{frame}

% =============================================================================
\begin{frame}{Contributions}
  \begin{enumerate}
    \item \hl{Novel task}: region-conditioned standard$\rightarrow$dialect generation (Feature~C) for all three regions.
    \item \hl{Unified multi-task model}: one BARTpho-syllable does (A) dialect$\rightarrow$std,
          (B) lexical normalization, (C) std$\rightarrow$dialect --- selected by a text task prefix.
    \item \hl{Evaluation suite for one-to-many}: Dialect Feature Recall, marker-independent
          Dialectal Edit-Recall, region accuracy, copy-rate (BLEU alone is misleading here).
    \item \hl{Data strategy}: oversampling $+$ sampling back-translation for the data-starved regions.
  \end{enumerate}
\end{frame}

\begin{frame}{Method (1): multi-task task-prefix model}
  \centering
  \begin{tikzpicture}[font=\small,node distance=6mm,
      box/.style={draw,rounded corners,minimum height=0.95cm,align=center}]
    \node[box,fill=oigray!12,minimum width=4.3cm] (in) {\texttt{\{task\}: \{source\}}};
    \node[box,fill=oiblue!15,right=of in,minimum width=1.9cm] (enc) {Encoder};
    \node[box,fill=oiblue!15,right=of enc,minimum width=1.9cm] (dec) {Decoder};
    \node[box,fill=oiorange!18,right=of dec,minimum width=2.6cm] (out) {target\\sequence};
    \draw[-{Latex[length=2mm]}] (in) -- (enc);
    \draw[-{Latex[length=2mm]}] (enc) -- (dec);
    \draw[-{Latex[length=2mm]}] (dec) -- (out);
    \node[below=1mm of enc,font=\scriptsize,text=oiblue] {\textbf{one BARTpho-syllable}};
  \end{tikzpicture}
  \vspace{0.2em}
  All three tasks are one conditional generation problem $P_\theta(y\mid x,c)=\prod_{t}P_\theta(y_t\mid y_{<t},x,c)$, source $x$, prefix-condition $c$:
  \begin{itemize}\small
    \item \hl{Textual conditioning}: input is \texttt{\{task\}: \{source\}} --- \hl{no new heads}, all tasks share parameters and tokenizer. Prefixes: \tprefix{dialect2std}, \tprefix{std2dialect\_\{north,central,south\}}, \tprefix{lexnorm}.
    \item \hl{BARTpho-syllable}: syllable tokenization is robust to dialect/teencode OOV (no fragile word segmenter); expect \hl{positive transfer} among the three noisy-input tasks.
    \item \hl{Train} teacher forcing $+$ \hl{label smoothing}; \hl{decode} beam (deterministic normalization) vs.\ \emph{sampling} (generation $+$ back-translation, avoids mode collapse --- Edunov et al.\ 2018).
  \end{itemize}
\end{frame}

% =============================================================================
\begin{frame}{Method (2): reverse corpus + sampling back-translation}
  \centering
  \begin{tikzpicture}[font=\scriptsize,node distance=5mm and 8mm,
      box/.style={draw,rounded corners,minimum height=0.8cm,align=center,inner sep=3pt}]
    \node[box,fill=oiorange!18] (rd) {real \textbf{dialect}\\(target)};
    \node[box,fill=oiblue!15,right=of rd] (am) {A-model\\dialect$\to$std\\(\emph{sample})};
    \node[box,fill=oigray!12,right=of am] (ss) {synthetic\\\textbf{standard} (source)};
    \node[box,fill=oigreen!15,right=of ss,minimum width=2.4cm] (pair) {train pair for C:\\(synth-std, real-dia)};
    \draw[-{Latex[length=2mm]}] (rd) -- (am);
    \draw[-{Latex[length=2mm]}] (am) -- (ss);
    \draw[-{Latex[length=2mm]}] (ss) -- (pair);
    \draw[-{Latex[length=2mm]},oiorange,thick] (rd.south) to[bend right=25] (pair.south);
  \end{tikzpicture}
  \vspace{0.3em}
  \begin{itemize}\small
    \item \hl{Reverse corpus}: swap (dialect, standard) $\rightarrow$ (standard, dialect) $+$ region prefix.
          Target side stays \emph{human-authored}.
    \item \hl{Sampling back-translation} (Edunov et al.\ 2018): keep \emph{real dialect} as target,
          \emph{sample} synthetic standard sources from the strong A-model --- sampling, not beam, to avoid mode collapse.
    \item \hl{Rebalancing}: oversample Northern/Southern; light quality filter (round-trip, length ratio, $\geq$1 marker).
    \item \hl{Transfer-aware selection}: pooled val BLEU rewards a \emph{copying} checkpoint, so we select on a \hl{macro per-task BLEU $+$ Dialectal Edit-Recall} composite --- preferred only if it actually performs the dialectal substitutions.
  \end{itemize}
\end{frame}

% =============================================================================
\begin{frame}{Evaluation metrics}
  \begin{itemize}\small
    \item \textbf{Overlap/semantic}: BLEU, chrF, ROUGE-L, METEOR, BERTScore;
          \textbf{normalization}: WER/CER (A), ERR (B).
    \item \textbf{Dialect Feature Recall (DFR)} --- fraction of the reference's regional markers reproduced:
      \[
        \mathrm{DFR}=\frac{\bigl|\{m\in R: m\in\hat y\}\bigr|}{|R|},\quad R=\{m\in M_r: m\in y\}.
      \]
    \item \textbf{Dialectal Edit-Recall} (marker-independent) --- did the model make the gold's token changes?
      \[
        \mathrm{Recall}=\frac{|A_{\text{gold}}\cap A_{\text{sys}}|}{|A_{\text{gold}}|},\quad
        A_\bullet = T(\cdot)\setminus T(x).
      \]
      Copy $\Rightarrow$ Edit-Recall $=0$, regardless of BLEU.
    \item \textbf{Region accuracy} $+$ \textbf{copy-rate}: controllability and under-generation.
  \end{itemize}
\end{frame}

\begin{frame}{Analysis \& error taxonomy + limitations}
  \begin{itemize}\small
    \item \hl{Under-transfer / copying} (copy-rate, Edit-Recall) --- worst on Northern.
    \item \hl{Wrong-region leakage} (region accuracy) and \hl{over-transfer} (Edit-Precision, spurious markers); \hl{semantic drift} (BERTScore / human eval).
    \item \hl{Diversity}: sampling yields varied valid realizations (Self-BLEU Central 81.6, Southern 86.7, Northern 93.7; lower $=$ more diverse) --- realizations a single-reference BLEU would penalize.
  \end{itemize}
  \vspace{0.2em}
  \begin{block}{Limitations \& future work}\small
    Single dialectal reference per sentence; 63 provinces collapsed to 3 regions; DFR depends on a marker list (mitigated by marker-independent Edit-Recall); Northern smallest with synthetic back-translation sources. \emph{Future}: human evaluation at scale, ViT5 comparison, iterative back-translation.
  \end{block}
\end{frame}

\begin{frame}{Conclusion}
  \begin{itemize}
    \item First \hl{region-conditioned standard$\rightarrow$dialect} generator for Vietnamese.
    \item One multi-task model: normalization \emph{and} generation \emph{and} lexical normalization.
    \item Normalization competitive with a much larger SOTA model.
    \item New transfer-aware metrics expose what BLEU hides; data strategy fixes regional imbalance.
  \end{itemize}
